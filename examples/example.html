Examples

Inflated,

$('.default h1').inflateText();

scaled to 80%, scaled to 30%, 


      $('.scale h1 .one').inflateText({ scale:0.8 });
      $('.scale h1 .two').inflateText({ scale:0.3 });

at least 76px

$('.min h1').inflateText({minFontSize:76});

..and no more than 32px

$('.max h1').inflateText({maxFontSize:32});

via AMD


require(['../inflateText.js'], function (inflateText) {
  inflateText(document.querySelector('.amd-default h1'));
});
